function parseDate(d){

  if(!d) return '';

  const dt = new Date(d);

  if(isNaN(dt)) return d;

  return dt.toLocaleDateString('pt-BR',{day:'2-digit',month:'short',year:'numeric'});

}


function stripHtml(h){ return h ? h.replace(/<[^>]*>/g,'').replace(/&[a-z]+;/g,' ').trim().slice(0,180) : ''; }


function getCatFromTags(tags=[], title=''){

  const t = (tags.join(' ')+' '+title).toLowerCase();

  if(/festival|cannes|berlin|venice|sundance|tribeca|rotterdam/.test(t)) return 'Festival';

  if(/oscar|award|premi|indica|shortlist|nomina/.test(t)) return 'Awards';

  if(/lança|estreia|release|estreou/.test(t)) return 'Lançamento';

  if(/crítica|review|resenha|análise/.test(t)) return 'Crítica';

  return 'Mercado';

}


// Try all RSS feeds for a panel, return merged items (up to 6)

async function fetchRSS(panel){

  const feeds = FEEDS[panel];

  const allItems = [];

  await Promise.allSettled(feeds.map(async ({url, fonte})=>{

    try{

      const res = await fetch(`${RSS2JSON}?rss_url=${encodeURIComponent(url)}&count=4`);

      if(!res.ok) throw new Error(`HTTP ${res.status}`);

      const data = await res.json();

      if(data.status !== 'ok' || !data.items?.length) throw new Error('No items');

      data.items.forEach(item=>{

        allItems.push({

          categoria: getCatFromTags(item.categories||[], item.title||''),

          titulo: item.title||'—',

          resumo_2_linhas: stripHtml(item.description||item.content||''),

          data: parseDate(item.pubDate),

          fonte: fonte,

          url: item.link||null

        });

      });

    }catch(e){ console.warn(`RSS failed (${fonte}):`, e.message); }

  }));

  // Sort by date desc, deduplicate by title, take 6

  allItems.sort((a,b)=> new Date(b.data)-new Date(a.data));

  const seen = new Set();

  return allItems.filter(i=>{ if(seen.has(i.titulo)) return false; seen.add(i.titulo); return true; }).slice(0,6);

}


// Fallback: Anthropic via Netlify proxy (no key exposed)

async function fetchAnthropicFallback(panel){

  // In Claude.ai preview, call Anthropic directly; on Netlify, use /api/anthropic proxy

  const isNetlify = !window.location.hostname.includes('claude.ai') && window.location.hostname !== 'localhost';

  const endpoint = isNetlify ? '/api/anthropic' : 'https://api.anthropic.com/v1/messages';

  const extraHeaders = isNetlify ? {} : {'anthropic-dangerous-direct-browser-access':'true'};


  const res = await fetch(endpoint,{

    method:'POST',

    headers:{'Content-Type':'application/json','anthropic-version':'2023-06-01',...extraHeaders},

    body:JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:1000,tools:[{"type":"web_search_20250305","name":"web_search"}],messages:[{role:'user',content:PROMPTS[panel]}]})

  });

  if(!res.ok) throw new Error(`HTTP ${res.status}`);

  const data = await res.json();

  const rawText = data.content.filter(b=>b.type==='text').map(b=>b.text).join('');

  const stripped = rawText.replace(/```(?:json)?\s*/gi,'').replace(/```/g,'').trim();

  const arrStart=stripped.indexOf('['), arrEnd=stripped.lastIndexOf(']');

  if(arrStart===-1||arrEnd===-1) throw new Error('No array');

  const items = JSON.parse(stripped.slice(arrStart,arrEnd+1));

  if(!Array.isArray(items)||!items.length) throw new Error('Empty');

  return items;

}


function renderItems(items, newsEl){

  newsEl.innerHTML='';

  newsEl.style.display='grid';

  items.forEach(item=>{

    const catClass=getCatClass(item.categoria||'default');

    const card=document.createElement('div');

    card.className='news-card';

    const url=item.url&&item.url.startsWith('http')?item.url:null;

    card.innerHTML=`

      <span class="cat-badge cat-${catClass}">${item.categoria||'News'}</span>

      <h4>${item.titulo||'—'}</h4>

      <p>${item.resumo_2_linhas||''}</p>

      <div class="news-card-footer">

        <span class="news-date">${item.data||''}</span>

        <span class="news-source">${url?`<a href="${url}" target="_blank" rel="noopener">${item.fonte||'Fonte'}</a>`:`<span style="font-size:.6rem;letter-spacing:.1em;text-transform:uppercase;color:var(--silver);opacity:.5">${item.fonte||''}</span>`}</span>

      </div>`;

    newsEl.appendChild(card);

  });

}


async function fetchPanel(panel){

  const skelEl=document.getElementById(`skel-${panel}`);

  const newsEl=document.getElementById(`news-${panel}`);

  const errEl=document.getElementById(`error-${panel}`);

  skelEl.style.display='grid'; newsEl.style.display='none'; errEl.style.display='none';

  newsEl.innerHTML='';

  try{

    // Try RSS first

    let items = await fetchRSS(panel);

    if(items.length >= 2){

      skelEl.style.display='none';

      renderItems(items, newsEl);

      return;

    }

    // Fallback to Anthropic if RSS got < 2 items

    console.log(`RSS returned only ${items.length} items for ${panel}, trying Anthropic fallback...`);

    items = await fetchAnthropicFallback(panel);

    skelEl.style.display='none';

    renderItems(items, newsEl);

  }catch(e){

    console.error(e);

    skelEl.style.display='none';

    errEl.style.display='flex';

  }

}


function getCatClass(cat){

  const c=(cat||'').toLowerCase();

  if(c.includes('festival')) return 'festival';

  if(c.includes('award')||c.includes('premia')) return 'awards';

  if(c.includes('lança')||c.includes('release')||c.includes('estreia')) return 'lançamento';

  if(c.includes('critic')||c.includes('review')) return 'crítica';

  return 'default';

}
